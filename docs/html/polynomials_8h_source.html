140  const unsigned int &iSymbol) {

142 
143  // Removing eventual 0
144  while (iPolynomial[iPolynomialDegree] == 0)
145  iPolynomialDegree--;
146 
147  vector<Type> iWorking(iPolynomial.begin(),
148  iPolynomial.begin() + iPolynomialDegree + 1);
149  vector<Type> iQuotient;
150

152  Type iTemp;
153 
154  if (iPolynomialDegree < iSymbol - 1)
155  return false;
156 
157  while (iPolynomialDegree >= iSymbol) {

159  iQuotient.insert(iQuotient.begin(), iTemp);

163 
164  iPolynomialDegree--;
165

167  iQuotient.insert(iQuotient.begin(), 0);

171 
172  if (iPolynomialDegree < iSymbol - 1) {

174  if (iWorking[i] != 0)
175  return false;
176  }
177  }
178

200 template <typename Type>
201 bool dividePolynomialByPolynomial(vector<Type> &iNumerator,
202  const vector<Type> &iDenominator) {
203  unsigned int dN(iNumerator.size() - 1), dD(iDenominator.size() - 1);
204 
205  if (dN < dD || (dD == 0 && iDenominator[0] != 0))
206  return false;
207 
208  vector<Type> iWorking(iNumerator), iQuotient;
209 
210  unsigned int i;
211  Type iTemp;
212 
213  while (dN >= dD) {
214  if (iWorking[dN] % iDenominator[dD] != 0)
215  return false;
216 
217  iTemp = iWorking[dN] / iDenominator[dD];
218  iQuotient.insert(iQuotient.begin(), iTemp);
219 
220  for (i = 0; i <= dD; i++)
221  iWorking[dN - i] -= iTemp * iDenominator[dD - i];
222 
223  dN--;
224 
225  while (iWorking[dN] == 0 && dN >= 1 && dN >= dD) {
226  iQuotient.insert(iQuotient.begin(), 0);
227  dN--;
228  }
229  }
230 
231  for (i = 0; i <= dN; i++) {
232  if (iWorking[i] != 0)
233  return false;
234  }
235 
236  iNumerator = iQuotient;
237 
238  return true;
