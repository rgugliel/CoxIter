64 iSQRT(Type iN) {
65  Type place =
66  (Type)1
67  << (sizeof(Type) * 8 -
68  2); // calculated by precompiler = same runtime as: place = 0x40000000
69  while (place > iN)
70  place /= 4; // optimized by complier as place >>= 2
71 
72  Type root = 0;
73 
74  while (place) {
75  if (iN >= root + place) {
76  iN -= root + place;
77  root += place * 2;
78  }
79 
80  root /= 2;
81  place /= 4;
82  }
83 
84  return root;
85 }
