967 inline unsigned int iLinearizationMatrix_index(const unsigned int &i,
968  const unsigned int &j,

973 inline unsigned int iLinearizationMatrix_row(const unsigned int &k,
974  const unsigned int &n) {
975  return ((2 * n + 1 - iSQRTsup((2 * n + 1) * (2 * n + 1) - 8 * k)) / 2);

978 inline unsigned int iLinearizationMatrix_col(const unsigned int &k,
